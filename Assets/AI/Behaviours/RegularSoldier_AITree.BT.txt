tree("Root")
	fallback
		tree("Stunned")
		tree("Combat")
		tree("WalkTheRoad")

tree("WalkTheRoad")
	while not inCombat
		GoToDestination


tree("Combat")
	while inCombat
		fallback
			while not CombatTargetAlive
				Fail
			while WithinCombatRange
				while CombatTargetAlive	
					parallel
						AttackTarget	
						RestoreStamina
			while EnoughStaminaToRun
				parallel
					SpeedUpToRun
					ChaseCombatTarget
					DepleteStamina
			while recoveringStamina
				sequence
					repeat
						parallel
							SlowDownToWalk
							RestoreStamina
							ChaseCombatTarget
			
		
tree("Stunned")
	while isStunned
		Succeed
