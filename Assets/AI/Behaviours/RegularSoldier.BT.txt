tree("Root")
	fallback
		tree("Stunned")
		tree("WalkTheRoad")
		tree("Combat")

tree("WalkTheRoad")
	while not inCombat
		GoToDestination


tree("Combat")
	while inCombat
		fallback
			while WithinCombatRange
				parallel
					AttackTarget	
					RestoreStamina
			while EnoughStaminaToRun
				parallel
					SpeedUpToRun
					ChaseCombatTarget
					DepleteStamina
			while recoveringStamina
				sequence
					repeat
						parallel
							SlowDownToWalk
							RestoreStamina
							ChaseCombatTarget
			
		
tree("Stunned")
	while isStunned
		Succeed
